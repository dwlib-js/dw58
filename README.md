# DW58

## Install
`npm i --save @dwlib/dw58`

## Usage
```javascript
// CJS
const DW58 = require('@dwlib/dw58');
const DW58Decode = require('@dwlib/dw58/decode');
const DW58DecodeBigInt = require('@dwlib/dw58/decodeBigInt');
const DW58DecodeInt = require('@dwlib/dw58/decodeInt');
const DW58DecodeInto = require('@dwlib/dw58/decodeInto');
const DW58Encode = require('@dwlib/dw58/encode');
const DW58EncodeBigInt = require('@dwlib/dw58/encodeBigInt');
const DW58EncodeInt = require('@dwlib/dw58/encodeInt');
const DW58IsValid = require('@dwlib/dw58/isValid');
const DW58Normalize = require('@dwlib/dw58/normalize');
const DW58Validate = require('@dwlib/dw58/validate');
// ESM
import DW58 from '@dwlib/dw58';
import DW58Decode from '@dwlib/dw58/decode';
import DW58DecodeBigInt from '@dwlib/dw58/decodeBigInt';
import DW58DecodeInt from '@dwlib/dw58/decodeInt';
import DW58DecodeInto from '@dwlib/dw58/decodeInto';
import DW58Encode from '@dwlib/dw58/encode';
import DW58EncodeBigInt from '@dwlib/dw58/encodeBigInt';
import DW58EncodeInt from '@dwlib/dw58/encodeInt';
import DW58IsValid from '@dwlib/dw58/isValid';
import DW58Normalize from '@dwlib/dw58/normalize';
import DW58Validate from '@dwlib/dw58/validate';
```
